% This document demonstrates Task 4:
% Customizing the Table of Contents (ToC), List of Figures (LoF),
% List of Tables (LoT), and creating an index.

\documentclass[11pt, a4paper]{article}

% --- UNIVERSAL PREAMBLE BLOCK ---
\usepackage[a4paper, top=2.5cm, bottom=2.5cm, left=2cm, right=2cm]{geometry}
\usepackage{fontspec}
\usepackage[english, bidi=basic, provide=*]{babel}
\babelprovide[import, onchar=ids fonts]{english}
\babelfont{rm}{Noto Sans}
% --- END UNIVERSAL PREAMBLE BLOCK ---

% 'tocloft' is a powerful package for customizing the ToC, LoF, and LoT
\usepackage{tocloft}

% 'imakeidx' simplifies index generation (requires no extra steps)
\usepackage{imakeidx}
\makeindex % Command to enable indexing

% --- ToC Customizations ---
% 1. Add dots as leaders for section entries
\renewcommand{\cftsecleader}{\cftdotfill{\cftdotsep}}
% 2. Change the font for section titles in the ToC to bold
\renewcommand{\cftsecfont}{\bfseries}
% 3. Increase the space for section numbers
\setlength{\cftsecnumwidth}{3em}
% --- End Customizations ---


\begin{document}

% These commands generate the lists
\tableofcontents
\listoffigures
\listoftables

\clearpage

\section{First Section}

This section contains important keywords. Here is one keyword.\index{Keyword}
We can also index entire phrases.\index{Important Phrase}

\begin{figure}[htbp]
  \centering
  \framebox{This is a placeholder for a figure.}
  \caption{A Sample Figure}
  \label{fig:sample}
\end{figure}

\section{Second Section}

This section has another keyword.\index{Another Keyword}
We can also create sub-entries in the index.\index{Keyword!sub-entry}

\begin{table}[htbp]
  \centering
  \framebox{This is a placeholder for a table.}
  \caption{A Sample Table}
  \label{tab:sample}
\end{table}

This text references the figure (Figure~\ref{fig:sample}) and the table (Table~\ref{tab:sample}).

% This command generates the index at this spot
\printindex

\end{document}
