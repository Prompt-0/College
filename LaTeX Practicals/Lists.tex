% Creating and customizing bulleted, numbered, and definition lists.
% This includes compact lists, custom labels, and resuming lists.

\usepackage[a4paper, top=2.5cm, bottom=2.5cm, left=2cm, right=2cm]{geometry}

\babelprovide[import, onchar=ids fonts]{english}

% 'enumitem' is the best package for customizing lists
\usepackage{enumitem}

\begin{document}

\section{Standard Lists}

\subsection{Bulleted List (Itemize)}
\begin{itemize}
    \item First item
    \item Second item
    \item Third item
\end{itemize}

\subsection{Numbered List (Enumerate)}
\begin{enumerate}
    \item First item
    \item Second item
    \item Third item
\end{enumerate}

\subsection{Definition List (Description)}
\begin{description}
    \item[LaTeX] A document preparation system.
    \item[Itemize] The environment for bulleted lists.
\end{description}

\section{Customized Lists}

\subsection{Compact List (no spacing)}
Using \texttt{[nosep]} from \texttt{enumitem} removes all vertical spacing.
\begin{itemize}[nosep]
    \item Compact item
    \item Compact item
    \item Compact item
\end{itemize}

\subsection{Customized Labels}
We can change the numbering style, e.g., to Roman numerals.
\begin{enumerate}[label=\Roman*.]
    \item A list item with a Roman numeral.
    \item Another item.
    \item \texttt{label=\Roman*.} controls the style.
\end{enumerate}

\subsection{Interrupting and Resuming a List}
Here is the first part of a list:
\begin{enumerate}[label=(\arabic*)]
    \item Item one.
    \item Item two.
\end{enumerate}
Now, some interrupting text is placed between the list segments.
\begin{enumerate}[resume]
    \item Item three, which continues the numbering.
    \item Item four.
\end{enumerate}
